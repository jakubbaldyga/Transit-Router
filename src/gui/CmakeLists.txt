include(FetchContent)

if(MINGW AND CMAKE_CXX_COMPILER_ID STREQUAL "Clang")
    # MinGW libstdc++ uses emulated TLS for std::once internals.
    add_compile_options(-femulated-tls)
    add_link_options(-femulated-tls)
endif()

# SFML
FetchContent_Declare(
    sfml
    GIT_REPOSITORY https://github.com/SFML/SFML.git
    GIT_TAG 3.0.1
)
set(SFML_BUILD_AUDIO OFF CACHE BOOL "" FORCE)
set(SFML_USE_STATIC_STD_LIBS OFF CACHE BOOL "" FORCE)
FetchContent_MakeAvailable(sfml)

find_package(Threads REQUIRED)

set(BUILD_SHARED_LIBS OFF CACHE BOOL "" FORCE)
set(BUILD_CURL_EXE OFF CACHE BOOL "" FORCE)
set(CURL_USE_LIBPSL OFF CACHE BOOL "" FORCE)
set(CURL_ZLIB OFF CACHE BOOL "" FORCE)
set(CURL_USE_LIBIDN2 OFF CACHE BOOL "" FORCE)
set(CURL_ENABLE_SSL ON CACHE BOOL "" FORCE)
set(CURL_USE_SCHANNEL ON CACHE BOOL "" FORCE)
set(ENABLE_MANUAL OFF CACHE BOOL "" FORCE)
set(BUILD_TESTING OFF CACHE BOOL "" FORCE)

FetchContent_Declare(
    curl
    GIT_REPOSITORY https://github.com/curl/curl.git
    GIT_TAG curl-8_19_0
)
FetchContent_MakeAvailable(curl)


set(CPR_USE_EXISTING_CURL_TARGET ON CACHE BOOL "" FORCE)
set(CPR_CURL_USE_LIBPSL OFF CACHE BOOL "" FORCE)
set(CPR_ENABLE_CURL_HTTP_ONLY OFF CACHE BOOL "" FORCE)
FetchContent_Declare(
    cpr
    GIT_REPOSITORY https://github.com/libcpr/cpr.git
    GIT_TAG 1.14.2 
)
FetchContent_MakeAvailable(cpr)

file(GLOB_RECURSE GUI_SOURCES "*.cpp")

add_executable(gui ${GUI_SOURCES})

target_include_directories(gui PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})
target_link_libraries(gui PRIVATE
    core
    SFML::Graphics
    SFML::Window
    SFML::System
    cpr::cpr
    CURL::libcurl
    Threads::Threads
)

if(MINGW AND CMAKE_CXX_COMPILER_ID STREQUAL "Clang")
    # Clang+MinGW with lld may miss libstdc++ internals used by std::call_once.
    target_link_libraries(gui PRIVATE stdc++)
endif()

copy_core_dll(gui)
